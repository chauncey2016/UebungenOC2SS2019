\section{Generator für TSP-Probleme}

\begin{frame}{Generator für TSP-Problem}
    \begin{itemize}
    \item Abstandsmatrix $M \in {\R}^{u^{2}}$
    \item Kosten $v,w \in [0,u]$
    \item $dst(v,w) = M[v*u+w] = M[w*u+v] \Rightarrow$ 
    doppelt so groß wie nötig
    \end{itemize}
\end{frame}
